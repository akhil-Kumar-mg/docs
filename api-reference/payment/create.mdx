---
title: 'Create Order'
api: 'POST /pg/orders'
description: 'Create a new order for payment processing.'
---

<RequestExample>
```bash cURL
curl --request POST \
  --url https://api.eximpe.com/pg/orders \
  --header 'Content-Type: application/json' \
  --header 'X-Client-Id: YOUR_CLIENT_ID' \
  --header 'X-Client-Secret: YOUR_CLIENT_SECRET' \
  --data '{
    "amount": "100.00",
    "currency": "INR",
    "reference_id": "order_123456",
    "return_url": "https://your-website.com/payment-success",
    "mop_type": "UPI",
    "buyer": {
      "name": "John Doe",
      "email": "john.doe@example.com",
      "phone": "+919876543210",
      "address": {
        "line_1": "123 Main Street",
        "line_2": "Apartment 4B",
        "city": "Mumbai",
        "state": "Maharashtra",
        "postal_code": "400001"
      }
    },
    "product": {
      "name": "Premium Subscription",
      "description": "Monthly premium subscription plan",
      "hs_code": "85287200",
      "hs_code_description": "Digital content and services",
      "type_of_goods": "goods"
    },
    "invoice": {
      "number": "INV-2024-001",
      "date": "2024-01-15"
    }
  }'
```
</RequestExample>

## Headers

<ParamField header="X-Client-Id" type="string" required>
  Your API client ID
</ParamField>

<ParamField header="X-Client-Secret" type="string" required>
  Your API client secret
</ParamField>

## Body

<ParamField body="amount" type="string" required>
  Amount in decimal format (e.g., "100.00")
</ParamField>

<ParamField body="currency" type="string" required>
  3-letter ISO currency code (e.g., INR, USD)
</ParamField>

<ParamField body="reference_id" type="string" required>
  Unique identifier for the order
</ParamField>

<ParamField body="buyer" type="object" required>
  The buyer details that are necessary. Note that all fields are required except email and phone.

  <Expandable title="Show child attributes">
    <ParamField body="buyer.name" type="string" required>
      Buyer's full name
    </ParamField>

    <ParamField body="buyer.email" type="string">
      Buyer's email address
    </ParamField>

    <ParamField body="buyer.phone" type="string">
      Buyer's phone number with country code
    </ParamField>

    <ParamField body="buyer.address" type="object" required>
      Buyer's address information

      <Expandable title="Show child attributes">
        <ParamField body="buyer.address.line_1" type="string" required>
          Address line 1
        </ParamField>

        <ParamField body="buyer.address.line_2" type="string">
          Address line 2
        </ParamField>

        <ParamField body="buyer.address.city" type="string" required>
          City name
        </ParamField>

        <ParamField body="buyer.address.state" type="string" required>
          State name
        </ParamField>

        <ParamField body="buyer.address.postal_code" type="string" required>
          Postal/ZIP code
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="product" type="object" required>
  Product information details

  <Expandable title="Show child attributes">
    <ParamField body="product.name" type="string" required>
      Product name
    </ParamField>

    <ParamField body="product.description" type="string">
      Product description
    </ParamField>

    <ParamField body="product.hs_code" type="string">
      Harmonized System code
    </ParamField>

    <ParamField body="product.hs_code_description" type="string">
      Description of the HS code
    </ParamField>

    <ParamField body="product.type_of_goods" type="string" required>
      Type of goods (e.g., goods)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="return_url" type="string">
  URL to redirect after payment
</ParamField>

<ParamField body="mop_type" type="string">
  Method of payment type (e.g., UPI)
</ParamField>

<ParamField body="invoice" type="object">
  Invoice information details

  <Expandable title="Show child attributes">
    <ParamField body="invoice.number" type="string">
      Invoice number
    </ParamField>

    <ParamField body="invoice.date" type="string">
      Invoice date in YYYY-MM-DD format
    </ParamField>
  </Expandable>
</ParamField>
